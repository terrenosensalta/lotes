/* ── PANNELLUM instances ── */
const viewers = {};

/* Mapa de URLs de imagen por desarrollo y vista */
const imageMap = {
cerro: ['assets/img/360/cerro-360.jpg', 'assets/img/360/cerro-360-norte.jpg', 'assets/img/360/cerro-360-ciudad.jpg'],
refugio: ['assets/img/360/refugio-360.jpg', 'assets/img/360/refugio-360-arroyo.jpg','assets/img/360/refugio-360-bosque.jpg'],
puma: ['assets/img/360/puma-360.jpg', 'assets/img/360/puma-360-dique.jpg', 'assets/img/360/puma-360-serrana.jpg'],

};

/* ── Tab switch ── */
function switchDev(devId, btn) {
document.querySelectorAll('.viewer-panel').forEach(p => p.classList.remove('active'));
document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
document.getElementById('panel-' + devId).classList.add('active');
btn.classList.add('active');
}

/* ── Load pannellum ── */
function loadPano(devId, imgSrc) {
const ph = document.getElementById('ph-' + devId);
const btn = document.getElementById('btn-' + devId);
const cmp = document.getElementById('comp-' + devId);

// Hide placeholder & button
ph.style.display = 'none';
btn.style.display = 'none';
cmp.style.display = 'flex';

if (viewers[devId]) {
viewers[devId].destroy();
}

viewers[devId] = pannellum.viewer('pano-' + devId, {
type: 'equirectangular',
panorama: imgSrc,
autoLoad: true,
autoRotate: -3,
compass: false,
showFullscreenCtrl: true,
showZoomCtrl: true,
mouseZoom: true,
hfov: 100,
strings: {
loadingLabel: 'Cargando vista 360°...',
noPanoramaError: 'Imagen 360° no encontrada. Agregá tu foto en: ' + imgSrc,
fileAccessError: 'Acceso denegado. Verificá que el archivo existe en: ' + imgSrc,
},
onError: function() {
ph.style.display = 'flex';
ph.querySelector('.placeholder-sub').textContent =
'No se encontró la imagen. Guardá tu foto equirrectangular en: ' + imgSrc;
ph.querySelector('.placeholder-title').textContent = '⚠ Imagen no encontrada';
btn.style.display = 'flex';
btn.textContent = '↺ Reintentar';
cmp.style.display = 'none';
}
});
}

/* ── Thumbnail selection ── */
function selectThumb(el, devId, idx) {
const strip = el.parentElement;
strip.querySelectorAll('.thumb').forEach(t => t.classList.remove('active-thumb'));
el.classList.add('active-thumb');

const imgSrc = imageMap[devId][idx];
if (viewers[devId]) {
viewers[devId].loadScene(imgSrc);
} else {
loadPano(devId, imgSrc);
}
}
